firefox ()
{
	if [[ -z "$1" ]]
	then
		/cygdrive/c/Windows/System32/cmd.exe /c "START C:\PROGRA~1\MozillaFirefox\firefox.exe --disable-gpu" & echo "Process id: $!"
	else
		ACTUAL_FILE_NAME="$(echo "$PWD/$1" | /usr/bin/sed "s/\/cygdrive\/\([a-z]\)/\1\u/;" | /usr/bin/sed "s/^\/home/F:\/cygwin\/home/;")"
		echo "Opening: $ACTUAL_FILE_NAME"
		/cygdrive/c/Windows/System32/cmd.exe /c "START C:\PROGRA~1\MozillaFirefox\firefox.exe --disable-gpu \"$ACTUAL_FILE_NAME\"" & echo "Process id: $!"
	fi
	Ret=$?
	return $Ret
}
